  \section{4.9: Application: The Handshake Theorem}

  \begin{defn*}
    The total degree of a graph is the sum of the degrees of all the vertices of the graph.  
  \end{defn*}
  
  \begin{thmBox*}[Theorem 4.9.1: The Handshake Theorem]
    If $G$ is any graph, then the sum of the degrees of all the vertices of $G$ equals twice the number of edges of $G$. Specifically, if the vertices of $G$ are $v_1,v_2,\dots,v_n$, where $n$ is a positive integer, then
    \begin{align*}
      \textnormal{the total degree of } G&= \operatorname{deg}(v_1)+\operatorname{deg}(v_2)+\dots+\operatorname{deg}(v_n)\\
        &=2\cdot\parens{\textnormal{the number of edges of $G$}}
    \end{align*}
  \end{thmBox*}
  
  \begin{thmBox*}[Corollary 4.9.2]
    The total degree of a graph is even.
  \end{thmBox*}

  \begin{thmBox*}[Proposition 4.9.3]
    In any graph there is an even number of vertices of odd degree.
  \end{thmBox*}
