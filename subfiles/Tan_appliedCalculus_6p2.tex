  \begin{thmBox*}
    Let $u=g(x)$, where $g$ is differentiable on an interval, and let $f$ be continuous on the corresponding range of $g$. On that interval,
    \[\int \textcolor{black}{f'\big(
      \textcolor{lander_blue}{g(x)}
    \big)}\cdot
    \textcolor{lander_blue}{g'(x)\,dx}
    =\int f(\textcolor{lander_blue}{u})\,
    \textcolor{lander_blue}{du}\]
  \end{thmBox*}
  \pagebreak

  \begin{thmBox*}[Procedure: Substitution Rule (Change of Variables)]
    \begin{enumerate}
      \item Let $u=g(x)$, where $g(x)$ is part of the integrand, usually the ``inside function'' of a composite function $f\parens{g(x)\vphantom{^1}}$.
      \item Find $du=g'(x)\,dx$.
      \item Use the substitution $u=g(x)$ and $du=g'(x)\,dx$ to convert the \emph{entire} integral into one involving only $u$.
      \item Find the resulting integral
      \item Replace $u$ by $g(x)$ to obtain the final solution as a function of $x$.
    \end{enumerate}
  \end{thmBox*}
