  \newcommand{\examOne}{
    Mean: & $\overline{x}=\dfrac{x_1+x_2+\dots+x_n}{n}=\displaystyle\dfrac{1}{n}\sum_{i=1}^n x_i$\\
    Standard Deviation: & $\displaystyle s=\sqrt{\dfrac{\displaystyle\sum \left(x_i-\overline{x}\right)^2}{n-1}}$\\
    Variance: & $\displaystyle s^2=\dfrac{\displaystyle\sum \left(x_i-\overline{x}\right)^2}{n-1}$\\
    $z$-Score: & $\displaystyle z=\dfrac{x-\overline{x}}{s}$\\
    Interquartile Range: & $IQR=Q_3-Q_1$\\
  }
  \newcommand{\examTwo}{
    $z$-Score: & $\displaystyle z=\dfrac{x-\overline{x}}{s}$\\
    Correlation Coefficient: & $r=\dfrac{\sum z_x z_y}{n-1}$\\
    Regression line: & $y=a+bx$, where $b=r\dfrac{s_y}{s_x}$ and $a=\overline{y}-b\overline{x}$\\
    Standard Error: & $SE=\sqrt{\dfrac{p(1-p)}{n}}$\\
  }
  \newcommand{\examThree}{
    Standard error (proportions)& $SE=\sqrt{\dfrac{p(1-p)}{n}}$\\
    Confidence intervals (proportions)& $\hat{p}\pm m$ \tab where $m=z^*SE_{\textnormal{est}}$,\tab  $SE_{\textnormal{est}}=\sqrt{\dfrac{\hat{p}(1-\hat{p})}{n}}$\\
    $z$ test statistic (proportions)& $z=\dfrac{\hat{p}-p_0}{SE}$ \tab where $SE=\sqrt{\dfrac{p_0(1-p_0)}{n}}$\\[10pt]
    Standard error (means)& $SE=\dfrac{\sigma}{\sqrt{n}}$\\
    Confidence intervals (means)& $\overline{x}\pm m$ \tab where $m=t^*SE_{\textnormal{est}}$,\tab  $SE_{\textnormal{est}}=\dfrac{s}{\sqrt{n}}$\\
    $t$ test statistic (means)& $t=\dfrac{\overline{x}-\mu}{SE_{\textnormal{est}}}$ \tab where $SE_{\textnormal{est}}=\dfrac{s}{\sqrt{n}}$\\

  }
