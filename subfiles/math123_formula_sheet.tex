  \newcommand{\expoRules}{

  }
  \newcommand{\derivShortcuts}{
    Product rule: & $\mathllap{\dfrac{d}{dx}\Big[f(x)\cdot g(x)\Big]}=f'(x)g(x)+f(x)g'(x)$\\
    Quotient rule: & $\mathllap{\dfrac{d}{dx}\left[\dfrac{f(x)}{g(x)}\right]}=\dfrac{g(x)f'(x)-f(x)g'(x)}{\left[g(x)\right]^2}$\\
    Chain rule: & $\mathllap{\dfrac{d}{dx}\Big[f\big(g(x)\big)\Big]}=f'\big(g(x)\big)\cdot g'(x)$\\
  }
